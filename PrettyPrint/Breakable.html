def pretty_print(q)
  q.text("breakable")

  attributes = [
    ("force=true" if force?),
    ("indent=false" unless indent?)
  ].compact

  if attributes.any?
    q.text("(")
    q.seplist(attributes, -> { q.text(", ") }) do |attribute|
      q.text(attribute)
    end
    q.text(")")
  end
end

def format(q)
  space = [If, IfMod, Unless, UnlessMod].include?(q.parent.class)

  left = node.left
  right = node.right

  q.format(left) if left
  q.text(" ") if space
  q.text(operator)
  q.text(" ") if space
  q.format(right) if right
end
